/**
 * Represents a movable object in the game.
 * Extends DrawableObject and provides movement, gravity, collision detection, and status functions.
 */
class MovableObject extends DrawableObject { 
   speed = 0.15;
   otherDirection = false;
   speedY = 0;
   acceleration = 2.5;
   energy = 100;
   lastHit = 0;

   /**
    * Applies gravity to the object.
    * - If the object is dead, it falls slowly.
    * - If the object is alive, normal gravity is applied.
    * - Throwable objects stop at y = 345.
    */
   applyGravity() {
       setInterval(() => {
           if (this.isDead()) {
               this.speedY += this.acceleration * 0.5; 
               this.y += this.speedY;  
           } else {
               if (this.isAboveGround() || this.speedY > 0) {
                   this.y -= this.speedY; 
                   this.speedY -= this.acceleration; 
               } 
 
               // Charakter landet auf dem Boden, aber nur wenn er nicht tot ist
               if (!(this instanceof ThrowableObject) && this.y > 151 && !this.isDead()) {  
                   this.y = 151; 
                   this.speedY = 0;
               }
           }
 
           // Flaschen fallen bis y = 345 und bleiben dort
           if (this instanceof ThrowableObject && this.y > 345) {  
               this.y = 345; 
               this.speedY = 0; 
           }
 
       }, 1000 / 25); 
   }

   /**
    * Checks if the object is above the ground.
    * @returns {boolean} True if the object is above the ground, otherwise false.
    */
   isAboveGround() {
       if (this instanceof ThrowableObject) {
           return this.y < 345; 
       } else {
           return this.y < 151;
       }
   }
  
   /**
    * Checks if this object is colliding with another object.
    * @param {MovableObject} mo - The other object to check for collision.
    * @returns {boolean} True if collision occurs, otherwise false.
    */
   isColliding(mo) {
       return (
           this.x + this.width - 40 > mo.x && 
           this.y + this.height - 10 > mo.y && 
           this.x < mo.x + mo.width && 
           this.y < mo.y + mo.height + 10 
       );
   }

   /**
    * Reduces the object's energy when hit.
    * - If invincible, no damage is taken.
    * - Activates temporary invincibility.
    */
   hit() {
       if (this.isInvincible) {
           return; 
       }

       this.energy -= 20; 
       if (this.energy < 0) {
           this.energy = 0; 
       } else {
           this.lastHit = new Date().getTime(); 
           this.makeInvincible(); 
       }
   }

   /**
    * Makes the object temporarily invincible after getting hit.
    */
   makeInvincible() {
       this.isInvincible = true; 
       setTimeout(() => {
           this.isInvincible = false;
       }, 500); 
   }

   /**
    * Updates the last hit timestamp when a coin is collected.
    */
   hitCoin() {
       this.lastHit = new Date().getTime();
   }

   /**
    * Checks if the object is currently in a "hurt" state.
    * @returns {boolean} True if the object is recently hurt, otherwise false.
    */
   isHurt() {
       let timePassed = new Date().getTime() - this.lastHit; 
       timePassed = timePassed / 1000; 
       return timePassed < 0.4;
   }

   /**
    * Checks if the object is dead (energy == 0).
    * @returns {boolean} True if energy is 0, otherwise false.
    */
   isDead() {
       return this.energy === 0;
   }

   /**
    * Plays an animation from a given array of image paths.
    * @param {string[]} images - The array of image paths to animate.
    */
   playAnimation(images) {
       if (images && images.length > 0) {
           let i = this.currentImage % images.length; 
           let path = images[i];
           this.img = this.imageCache[path];
           this.currentImage++;
       }
   }

   /**
    * Moves the object to the right.
    */
   moveRight() {
       this.x += this.speed;
   }

   /**
    * Moves the object to the left.
    */
   moveLeft() {
       this.x -= this.speed;
   }
}
